import { Component, OnInit } from '@angular/core';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { AuthService } from '../../services/auth.service';
import { ToastrService } from 'ngx-toastr';

@Component({
  selector: 'app-recover',
  templateUrl: './recover.component.html',
  styleUrls: ['./recover.component.css']
})
export class RecoverComponent implements OnInit {

  constructor(private fb: FormBuilder,
    private auth: AuthService,
    private toastr:ToastrService) { }

  ngOnInit(): void {
  }

  authForm: FormGroup = this.fb.group({
    correo: [localStorage.getItem('email') || '', [Validators.required, Validators.email]],
  })
/**
 * Funcion para recuperar contraseña olvidada
 */
  recuperar(){
    this.auth.recuperarCuenta(this.authForm.value['correo'])
    this.toastr.success('','Correo enviado!',{
      timeOut:1500,
      closeButton:true
       })
  }

}

<h1>RECUPERAR CONTRASEÑA</h1>
<form [formGroup]="authForm" (ngSubmit)="recuperar()">
    <div class="d-flex flex-column align-items-center">
        <p class="text-muted">Se enviará un correo electrónico con el proceso de recuperación de la cuenta, si la cuenta existe</p>
        <input type="email" formControlName="correo" class="caja form-control" placeholder="Correo electrónico">
    </div>
    <div *ngIf="this.authForm.get('correo')?.errors?.['required'] && this.authForm.get('correo').touched" class="text-danger text-center" role="alert">
        (El correo no puede quedar vacío)
    </div>
    <div *ngIf="this.authForm.get('correo')?.errors?.['email'] && this.authForm.get('correo').touched" class="text-danger text-center" role="alert">
        (El correo no es válido)
    </div>
    <div class="d-flex flex-column align-items-center mt-5">
        <button type="submit" class="btn btn-primary mb-3" value="1" [disabled]="this.authForm.invalid">Recuperar contraseña</button>
    </div>
</form>
